Transform

transform: ; Html elementinin 2D veya 3D dnüştürme özellikelrini belirler.

transform-origin:

transform-origin: özelliği ile html elementinin merkezini belirtiriz. Aldığı parametreler center,top,right,bottom,left veya sayısal değer.
transform-origin:center	;
/* Html elementinin dönüşüm merkezini elemetin merkezi olarak atadık bu zaten varsayılan değerdir */

transform-origin:1000px 600px	;
/* Html elementinin merkezini belirler. */

transform-origin:left	;
/* Html elementinin dönüşüm merkezini elemetin sol köşesi olarak değiştirdik */

transform-style:

transform-style: dönüşüm stili iki farklı şekilde olabilir → flat ve preserve-3d ;
  • flat; → 2D görüntü sağlıyor.
  • preserve-3d; → 3D görüntü sağlıyor / Çok iyi bunu kullan 3D lerde /

backface-visibility:

Belirtilen Html elementinin dönüştürme ,geçiş veya animasyon işlemlerinde arka yüzeyinin görüntülenme veya gizlenme özelliklerini belirler. İki farklı parametre alabilir → visible ve hidden.
backface-visibility:visible;	
/* dönüştürme sırasında elementin arka yüzünü gösterir. */

backface-visibility:hidden;
/* dönüştürme sırasında elementin arka yüzünü gizler. */

translate();

Html elementini x ve y ekseni doğrultusunda konumunu değiştir.
transform:translate(10px,15px);
/* Html elementini 10px sağa  ve 15px yukarı kaydırır */

translateX();

Html elementini x ekseni doğrultusunda konumunu değiştir.
transform:translateX(10px);
/* Html elementinin 10px sağa kaydırır. */

translateY();

Html elementini y ekseni doğrultusunda konumunu değiştir.
transform:translateY(5px);
/* Html elementini 5px kadar yukarı kaydırır. */

translateZ();

Html elementini z ekseni doğrultusunda konumunu değiştir.
transform:translateZ(perspective(1000px)  translateZ(40px));
/* Html elementini 40px kadar sayfa içine gömer. */

translate3d();

Html elementinin x ekseni , y ve z ekseni doğrultusunda konumunu değiştir.
transform:perspective(1000px)  translate3d(0,0,-330px);
/* perspective varsayılan değeri 1000px	(x,y,z ekseni); */

skew();

skew() Html elementinin x ve y ekseni doğrultusunda büker.
transform:skew(10deg,10deg);
/* Html elementini x ekseni doğrultusunda 10 derece , y doğrultusu boyunca 10 derece büker */

skewX();

Html elementinin x ekseni doğrultusunda büker.
transform:skewX(10deg);
/* Html elementini x ekseni doğrultusunda 10 derece büker */

skewY();

Html elementinin y ekseni doğrultusunda büker.
transform:skewY(10deg);
/* Html elementini y doğrultusu boyunca 10 derece büker */

scale();

scale(); Html elementinin x ve y ekseni doğrultusunda boyutunu değiştirir. Varsayılan değeri 1’dir ,2 yazarsan boyutu 2 katı olur 0,5 yazarsan boyutu yarısı olur
transform:scale(2);
/* elementi 2 kat büyütür */

scaleX();

Html elementinin x ekseni doğrultusunda yani sağa ve sola doğru boyutunu değiştirir.
transform:scaleX(2);
/* elementi yatay 2 kat büyütür */

scaleY();

Html elementinin y ekseni doğrultusunda yani aşağıya ve yukarıya doğru boyutunu değiştirir.
transform:scaleY(2);
/* elementi dikey 2 kat büyütür */

scaleZ();

Html elementini z ekseni doğrultusunda boyutunu değiştirir.
transform:perspective(1000px) scaleZ(2) translateZ(-100px);
/* elementi 2 kat büyütür */

scale3d();

  • Html elementinin x,y ve z ekseni doğrultusunda boyutunu değiştirir.
transform:perspective(1000px)  scale3d(0.5 , 1 , 0.5);

rotate();

rotate(); Html elementinin saat yönü doğrultusunda döndürür.
transform:rotate(10deg);
/* Html elementini saat yönü doğrultusunda 10 derece döndürür. */

rotateX();

Html elementini x ekseni doğrultusunda içe doğru döndürür.
transform:rotateX(180deg);
/* Html elementini yatay olarak saat yönü doğrultusunda 180 derece döndürür. */

rotateY();

Html elementini y ekseni doğrultusunda içe doğru döndürür.
transform:rotateY(180deg);
/* Html elementini dikey olarak saat yönü doğrultusunda 180 derece döndürür. */

rotate3d();

Html elementini x,y ve z ekseni doğrultusunda döndürür.
transform:rotate3d(1,1,1,190deg);
/* Html elementini dikey ve yatay olarak saat yönü doğrultusunda 190 derece döndürür. */
